Q.7 Find the maximum out of three numbers.

       
const Max_out_of_three = (A,B,C) => {
  if((A===B) && (B===C) && (C===A)){
    return -1;
  }
  else {
    if(A>B && A>C){
      return A;
    }
    else if(B>A && B>C){
      return B;
    }

      return C;
    }
  }
    

};
